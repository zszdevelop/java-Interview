                              # 1.4 Maven添加第三方仓库地址（全局配置）

                              # 1.5 配置settings.xml

                              在profiles 中添加如下内容

                              <repository>
          <id>3rdnexus</id>          <url>http://192.168.0.1:8081/repository/3rd/</url>
          <release>
            <enable>true</enable>
          </release>
          <snapshots>
            <enable>true</enable>
          </snapshots>
        </repository>
  </profiles>

                              添加server

                              <server>
      <id>3rdnexus</id>      #此ID要与profile中的ID对应
      <username>test</username>
      <password>123456</password>
 </server>
